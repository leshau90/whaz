\begin{algorithm}
    \caption*{Perhitungan CF Setiap Penyakit}
    \begin{algorithmic}[1]
        \Procedure{Akumulasi CF}{$\mathcal Penyakit$} \Comment asumsi input adalah \emph{array} Penyakit yang tiap elementnya ada Gejala dengan setiap element nya nilai MD dan MB
        \For  { \textbf{each} $p \in \mathcal Penyakit $}
        % {Penyakit \textbf{in} Gejala2}
            \State p.MB = 0
            \State p.MD = 0
            \For{ \textbf{each} $g \in p.Gejala $}
                \State p.MB = p.MB + (g.MD * (1 - p.MB))
                \State p.MD = p.MD + (g.MB * (1 - p.MD))
            \EndFor
            \State penyakit.CF = MB - MD 
        \EndFor
        \State Urutkan Penyakit berdasarkan CF terbesar 
        \State \Return Penyakit \Comment penyakit dengan CF terbesar atau paling diyakini oleh sistem ada di atas
        \EndProcedure
    \end{algorithmic}
\end{algorithm}
